#define BLUE  "\x1B[34m"

int c = 0;

void* fnC(void* arg)
{
    int i;
    int thread_no = *((int *)arg);

    for(i=0;i<1000;i++)
    {
        c++;

        if (thread_no)
            printf(" %s%d", BLUE, c);
        else
            printf(" %s%d", RED, c);
    }

    pthread_exit(NULL);
}
